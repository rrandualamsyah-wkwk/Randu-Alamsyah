<script>
let output=document.getElementById("output");
let historyDisplay=document.getElementById("history");

function appendNumber(num){if(output.innerText=="0")output.innerText=num;else output.innerText+=num;}
function appendOperator(op){output.innerText+=op;}
function appendDot(){if(!output.innerText.includes("."))output.innerText+=".";}
function clearDisplay(){output.innerText="0";historyDisplay.innerText="";}
function backspace(){output.innerText=output.innerText.slice(0,-1)||"0";}
function calculate(){try{let result=eval(output.innerText);historyDisplay.innerText=output.innerText;output.innerText=result;}catch{output.innerText="Error";}}
function calculateSin(){let val=parseFloat(output.innerText);output.innerText=Math.sin(val*Math.PI/180).toFixed(6);}
function calculateCos(){let val=parseFloat(output.innerText);output.innerText=Math.cos(val*Math.PI/180).toFixed(6);}
function calculateTan(){let val=parseFloat(output.innerText);output.innerText=Math.tan(val*Math.PI/180).toFixed(6);}
</script>
