  <script>
    const form = document.getElementById("formPendaftaran");
    const tabelBody = document.querySelector("#tabelPendaftaran tbody");
    let data = [];
    let editIndex = -1;

    form.addEventListener("submit", (e) => {
      e.preventDefault();

      const nama = document.getElementById("nama").value.trim();
      const kelas = document.getElementById("kelas").value.trim();
      const gmail = document.getElementById("gmail").value.trim();
      const ekskul = document.getElementById("ekskul").value;

      if (!gmail.endsWith("@gmail.com")) {
        alert("Alamat email harus menggunakan domain @gmail.com");
        return;
      }

      if (editIndex === -1) {
        data.push({ nama, kelas, gmail, ekskul });
      } else {
        data[editIndex] = { nama, kelas, gmail, ekskul };
        editIndex = -1;
      }

      form.reset();
      tampilkanData();
    });

    function tampilkanData() {
      tabelBody.innerHTML = "";
      data.forEach((item, index) => {
        const row = document.createElement("tr");

        row.innerHTML = `
          <td>${index + 1}</td>
          <td>${item.nama}</td>
          <td>${item.kelas}</td>
          <td>${item.gmail}</td>
          <td>${item.ekskul}</td>
          <td>
            <button class="edit" onclick="editData(${index})">Edit</button>
            <button class="hapus" onclick="hapusData(${index})">Hapus</button>
          </td>
        `;

        tabelBody.appendChild(row);
      });
    }

    function hapusData(index) {
      if (confirm("Yakin ingin menghapus data ini?")) {
        data.splice(index, 1);
        tampilkanData();
      }
    }

    function editData(index) {
      const item = data[index];
      document.getElementById("nama").value = item.nama;
      document.getElementById("kelas").value = item.kelas;
      document.getElementById("gmail").value = item.gmail;
      document.getElementById("ekskul").value = item.ekskul;
      editIndex = index;
    }
  </script>
